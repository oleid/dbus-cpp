find_package(PkgConfig)
find_package(Boost 1.49 COMPONENTS system)
find_package(Threads)

pkg_check_modules(DBUS dbus-1)

include_directories(
  ${CMAKE_SOURCE_DIR}/include/dbus
  ${DBUS_INCLUDE_DIRS}
  ${Boost_INCLUDE_DIRS}
  )

add_library(
  dbus_cpp SHARED

  bus.cpp
  error.cpp
  service.cpp

  asio/executor.cpp
)

target_link_libraries(
  dbus_cpp

  ${Boost_LIBRARIES}
  ${DBUS_LIBRARIES}
)
